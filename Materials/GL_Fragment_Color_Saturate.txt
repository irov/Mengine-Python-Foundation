#version 330 core

uniform float u_factor;

in vec4 v_Col;

out vec4 fragColor;

void main(void)
{
    const vec3 REC709 = vec3(0.2126, 0.7152, 0.0722);

    vec4 color = v_Col;

    float l = dot(color.rgb, REC709);
    vec3 grayscale = vec3(l);

    vec3 saturated = mix(grayscale, color.rgb, u_factor);

    fragColor = vec4(saturated, color.a);
}
