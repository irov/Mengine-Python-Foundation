#include <metal_stdlib>
using namespace metal;

struct FragmentIn
{
    float4 position [[position]];
    half4  color;
    float2 uv0;
};

fragment half4 fragmentShader(FragmentIn in [[stage_in]],
                              texture2d<half> inSampler0 [[texture(0)]],
                              sampler sampler0 [[sampler(0)]],
                              constant float *uniforms [[buffer(1)]])
{
    half u_factor = half(uniforms[0]);
    half4 c0 = inSampler0.sample(sampler0, in.uv0);
    half g = dot(c0.xyz, half3(0.299h, 0.587h, 0.114h));
    half3 sc = mix(c0.xyz, half3(g), u_factor);
    return half4(sc * half3(in.color.xyz), c0.w * in.color.w);
}
