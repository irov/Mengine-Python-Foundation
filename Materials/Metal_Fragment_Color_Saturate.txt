#include <metal_stdlib>
using namespace metal;

struct FragmentIn
{
    float4 position [[position]];
    half4  color;
};

fragment half4 fragmentShader(FragmentIn in [[stage_in]],
                              constant float *uniforms [[buffer(1)]])
{
    half u_factor = half(uniforms[0]);
    half4 c = in.color;
    half g = dot(c.xyz, half3(0.2126h, 0.7152h, 0.0722h));
    half3 sc = mix(c.xyz, half3(g), u_factor);
    return half4(sc, c.w);
}
