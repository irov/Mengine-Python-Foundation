uniform float u_factor;

sampler2D tex0;

struct v2p {
    float4 position : POSITION;
    float4 color    : COLOR0;
    float2 tex0     : TEXCOORD0;
};

struct p2f {
    float4 color : COLOR0;
};

const float3 REC709 = float3(0.2126, 0.7152, 0.0722);

void main(in v2p IN, out p2f OUT)
{
    float4 color = tex2D(tex0, IN.tex0);

    float l = dot(color.rgb, REC709);
    float4 grayscale = float4(l, l, l, color.a);

    float4 saturated = lerp(grayscale, color, u_factor);

    OUT.color = IN.color * saturated;
}
